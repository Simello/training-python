## String operations

    >>> a = "asdf"
    >>> b = "zxcv"
    >>> a + b
    'asdfzxvc'

    >>> c = "Python"
    >>> len(c)
    6
    >>> c[2]
    't'
    >>> c[:2]
    'Py'
    >>> c[2:4]
    'th'
    >>> c[4:]
    'on'
    >>> "sadf adsf sadf fdsa".split()
    ['sadf', 'adsf', 'sadf', 'fdsa']
    >>> "  adsffdsa    ".strip()
    'adsffdsa'

## Identity

    >>> l1 = ['a', 'b', 'c', 'd']
    >>> id(l1)
    .....
    >>> l2 = l1 # two varibles pointing to one object
    >>> id(l2)
    .....
    >>> l1 is l2
    True
    >>> l2.append('e')
    >>> l1
    ['a', 'b', 'c', 'd', 'e']
    >>> l3 = list(l2) # new copy of l2, same as l2[:]
